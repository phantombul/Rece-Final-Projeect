70   export const getRecipes = ({
71     ingredients,
72     dislikes,
73     allergies,
74     searchMode,
75     offset,
76     sortOption,
77     sortDirection,
78   }: RecipesRequest) =>
79     recipesAPI
80       .get<RecipesResponse>('/recipes/complexSearch', {
81         params: {
82           includeIngredients:
83             searchMode === 'all' ? ingredients.join(',') : undefined, // all these ingredients must be present in the recipes returned
84   
85           // if some is selected, not all ingredients will be included.
86           // also, in any request, either includeIngredients is used, or query. cannot have both as they are mutually exclusive because of the radio button
87   
88           query: searchMode === 'some' ? ingredients.join(' ') : undefined,
89           excludeIngredients: dislikes.join(','),
90           intolerances: allergies.join(','),
91           instructionsRequired: true,
92           addRecipeInformation: true,
93           offset,
94           sort: sortOption,
95           sortDirection,
96           number: 20,
97         },
98       })
99       .then((res) => res.data);

101  const getRecipesByCuisine = (cuisine: string) =>

108          sort: 'popularity',
109          instructionsRequired: true,
110          addRecipeInformation: true,

112      })
113      .then(({ data }) => ({ cuisine, data: data.results }));
